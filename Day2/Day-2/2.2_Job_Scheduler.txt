Start
Define class Job
    Attributes: name, deadline, processingTime
    Constructor(name, deadline, processingTime)
        Set values
End class

Initialize priorityQueue as empty PriorityQueue sorted by earliest deadline

Function addJob(name, deadline, processingTime)
    Create new Job(name, deadline, processingTime)
    Add to priorityQueue
End function

Function processJobs()
    While priorityQueue not empty
        Remove job with highest priority (earliest deadline)
        Print "Processing job: " + job.name
End function

Main
    addJob("Job1", 2, 5)
    addJob("Job2", 1, 3)
    addJob("Job3", 3, 2)
    processJobs()
End
